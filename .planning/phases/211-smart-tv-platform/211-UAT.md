## Evidence

```bash
# Platform detection
curl -s "http://127.0.0.1:8001/tv/platform/detect"
# Response: {"platform":"android_tv","user_agent":"curl/8.16.0"}

# UI config for Tizen
curl -s "http://127.0.0.1:8001/tv/ui-config?platform=tizen"
# Response: {"safe_area_horizontal":90,"safe_area_vertical":45,"focus_highlight_color":"#e50914","font_scale":1.5,"item_spacing":30,"carousel_item_width":300,"carousel_item_height":450}

# Remote keys
curl -s "http://127.0.0.1:8001/tv/remote-keys"
# Response: {"keys":["up","down","left","right","select","back","home","menu","play","pause","rewind","forward","voice"]}

# Features for tvOS
curl -s "http://127.0.0.1:8001/tv/features?platform=tvos"
# Response: {"platform":"tvos","features":["streaming","playback_controls","search","multi_user","airplay"]}

# Device registration
curl -s -X POST "http://127.0.0.1:8001/tv/devices" -H "Content-Type: application/json" -H "X-User-Id: 1" \
  -d '{"platform": "tizen", "device_id": "tv123", "device_name": "Samsung Smart TV", "model": "UN65TU8000", "hdr_capable": true}'
# Response: {"session_id":"2e6d3da4-d33c-4411-a6a3-20224b75a8b6","device_id":"tv123","platform":"tizen","user_id":1,...}

# Voice command
curl -s -X POST "http://127.0.0.1:8001/tv/voice" -H "Content-Type: application/json" \
  -d '{"command": "play action movies", "language": "en-US"}'
# Response: {"understood":true,"action":"search","query":"play action movies"}

# Playback settings
curl -s -X POST "http://127.0.0.1:8001/tv/playback-settings" -H "Content-Type: application/json" \
  -d '{"platform": "tvos", "device_id": "tv123", "hdr_capable": true, "dolby_vision": true, "dolby_atmos": true}'
# Response: {"preferred_resolution":2160,"hdr_enabled":true,"dolby_vision_enabled":true,"dolby_atmos_enabled":true,"frame_rate_matching":true}
```
